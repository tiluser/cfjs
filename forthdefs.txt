\ Puts DUP, ROT, and HELLO together 
: DRH ( n1 n2 -- n2 n2 n1 )

\ Tests the BEGIN-UNTIL indefinite loop
: TESTBU ( n --)
    BEGIN 
        1 +
        DUP 10 >
    UNTIL ;

\ Will loop until 11 is on the stack
1 TESTBU
